CREATE DATABASE MovieFlixDB

USE MovieFlixDB

CREATE TABLE Usuario (
id int NOT NULL,
cpf int NOT NULL,
nome varchar(255),
email varchar(255),
senha varchar(255),
PRIMARY KEY (id)
);

CREATE TABLE Filme (
cod_filme int NOT NULL,
titulo varchar(255),
imagem int NOT NULL,
descricao varchar(255),
genero varchar(255,)
PRIMARY KEY (cod_filme)
);

CREATE TABLE FilmeFavoritado (
id int NOT NULL,
titulo varchar(255),
imagem int NOT NULL,
PRIMARY KEY (id)
);
